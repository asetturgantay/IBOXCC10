26   public class MainActivity extends AppCompatActivity implements BottomNavigationView.OnNavigationItemSelectedListener, Toolbar.OnMenuItemClickListener {
27   
28       DrawerLayout drawerLayout;
29       ActionBarDrawerToggle toggle;
30   
31       @Override
32       protected void onCreate(Bundle savedInstanceState) {
33           super.onCreate(savedInstanceState);
34           setContentView(R.layout.activity_main);
35   
36           drawerLayout = findViewById(R.id.drawer);
37   
38           toggle = new ActionBarDrawerToggle(this, drawerLayout, R.string.nav_open, R.string.nav_close);
39           drawerLayout.addDrawerListener(toggle);
40           toggle.syncState();
41           getSupportActionBar().setDisplayHomeAsUpEnabled(true);
42   
43           loadFragment(new HomeFragment());
44   
45           BottomNavigationView bottomNavigationView = findViewById(R.id.bottom_navigation);
46           bottomNavigationView.setOnNavigationItemSelectedListener(this);
47       }
48   
49       @Override
50       public boolean onOptionsItemSelected(MenuItem item) {
51           if(toggle.onOptionsItemSelected(item)) {
52               return true;
53           }
54   
55           switch (item.getItemId()) {
56               case R.id.action_quit:
57                   LayoutInflater inflater = (LayoutInflater) getSystemService(Context.LAYOUT_INFLATER_SERVICE);
58                   View customView = inflater.inflate(R.layout.custom_view, null);
59                   new MaterialStyledDialog.Builder(this)
60                           .setTitle("Выберите комнату")
61                           .withIconAnimation(true)
62                           .withDialogAnimation(true)
63                           .withDarkerOverlay(true)
64                           .setCustomView(customView)
65                           .setPositiveText("Выбрать")
66                           .onPositive(new MaterialDialog.SingleButtonCallback() {
67                               @Override
68                               public void onClick(@NonNull MaterialDialog dialog, @NonNull DialogAction which) {
69                                   Log.d("MaterialStyledDialogs", "OK!");
70                               }
71                           })
72                           .setNegativeText("Отмена")
73                           .onNegative(new MaterialDialog.SingleButtonCallback() {
74                               @Override
75                               public void onClick(@NonNull MaterialDialog dialog, @NonNull DialogAction which) {
76                                   Log.d("MaterialStyledDialogs", "Cancel!");
77                               }
78                           })
79                           .show();
80                   break;
81           }
82           return super.onOptionsItemSelected(item);
83       }
84   
85       private boolean loadFragment(Fragment fragment) {
86   
87           if (fragment != null) {
88               getSupportFragmentManager()
89                       .beginTransaction()
90                       .replace(R.id.fragment_container, fragment)
91                       .commit();
92               return true;
93           }
94           return false;
95       }
96   
97       @Override
98       public boolean onNavigationItemSelected(@NonNull MenuItem item) {
99           Fragment fragment = null;
100          switch (item.getItemId())
101          {
102              case R.id.action_home:
103                  fragment = new HomeFragment();
104                  break;
105              case R.id.action_settings:
106                  fragment = new SettingsFragment();
107                  break;
108              case R.id.action_menu:
109                  fragment = new MenuFragment();
110                  break;
111          }
112          return loadFragment(fragment);
113      }
114  
115      @Override
116      public boolean onMenuItemClick(MenuItem item) {
117          return false;
118      }
119  
120      @Override
121      public boolean onCreateOptionsMenu(Menu menu) {
122  
123          getMenuInflater().inflate(R.menu.top_nav_menu, menu);
124          return true;
125      }
